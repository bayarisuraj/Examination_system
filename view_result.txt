<script>
const searchInput  = document.getElementById('searchInput');
const courseSelect = document.getElementById('courseSelect');
const resultsBody  = document.getElementById('resultsBody');
const spinner      = document.getElementById('loadingSpinner');
const tableWrap    = document.getElementById('tableWrap');

let debounceTimer;

function showLoading(){
    spinner.style.display = 'block';
    tableWrap.style.display = 'none';
}
function hideLoading(){
    spinner.style.display = 'none';
    tableWrap.style.display = 'block';
}

function updateSummary(rows){
    const scores = [];
    rows.forEach(tr => {
        const scoreCell = tr.querySelector('td[data-label="Score"]');
        if(scoreCell){
            const val = parseFloat(scoreCell.textContent);
            if(!isNaN(val)) scores.push(val);
        }
    });
    document.getElementById('totalCount').textContent = scores.length || '0';
    if(scores.length > 0){
        const avg = (scores.reduce((a,b) => a+b, 0) / scores.length).toFixed(1);
        document.getElementById('avgScore').textContent  = avg;
        document.getElementById('highScore').textContent = Math.max(...scores);
        document.getElementById('lowScore').textContent  = Math.min(...scores);
    } else {
        document.getElementById('avgScore').textContent  = '—';
        document.getElementById('highScore').textContent = '—';
        document.getElementById('lowScore').textContent  = '—';
    }
}

function fetchResults(){
    const search = encodeURIComponent(searchInput.value.trim());
    const course = encodeURIComponent(courseSelect.value);
    showLoading();
    fetch(`fetch_results.php?search=${search}&course=${course}`)
        .then(res => {
            if(!res.ok) throw new Error('Server error');
            return res.text();
        })
        .then(html => {
            hideLoading();
            resultsBody.innerHTML = html;
            const rows = resultsBody.querySelectorAll('tr[data-index]');
            rows.forEach((tr, i) => {
                const numCell = tr.querySelector('td.row-num');
                if(numCell) numCell.textContent = i + 1;
            });
            updateSummary(resultsBody.querySelectorAll('tr'));
        })
        .catch(err => {
            hideLoading();
            resultsBody.innerHTML = `
                <tr>
                    <td colspan="8" class="text-center text-danger py-4">
                        <i class="bi bi-exclamation-triangle-fill me-2"></i>
                        Error loading results. Please try again.
                    </td>
                </tr>`;
            console.error(err);
        });
}

function clearFilters(){
    searchInput.value  = '';
    courseSelect.value = '';
    fetchResults();
}

searchInput.addEventListener('input', () => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(fetchResults, 350);
});
courseSelect.addEventListener('change', fetchResults);
fetchResults();
</script>
